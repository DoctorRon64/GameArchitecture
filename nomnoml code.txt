//Generate Biome
[BiomeGenerator |
	void ChangeOneRandomBlock(RandomLocation) {
    	if (IsThisValidTerrainType()) {
        	AddTileToBiome();
        	CheckAroundBlockForTerrainType();
    	}
	}

	bool IsThisValidTerrainType() {
    	if (allTiles In Biome <= MaxBiomeSize && Match TerrainType) {
        	return true;
    	} return false;
]
